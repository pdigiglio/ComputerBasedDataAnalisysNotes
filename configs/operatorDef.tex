% Declare some mathematical operators

% ---------------------------------------------------------------------------------------
% STATISTICS

% Error function
\DeclareMathOperator{\erf}{erf}
% Covariance
\DeclareMathOperator{\cov}{cov}

% ---------------------------------------------------------------------------------------
% ALGEBRA

% Span of a set
\DeclareMathOperator{\Span}{Span}
%\DeclareMathOperator{\Spam}{Spam}
% Sign function
\DeclareMathOperator{\sgn}{sgn}
% Trace
\DeclareMathOperator{\tr}{tr}
% Diagonal matrix
\DeclareMathOperator{\diag}{diag}

% ---------------------------------------------------------------------------------------
% DIFFERENTIAL OPERATORS

% Nabla
\newcommand{\V}{\mathop{}\!\nabla}
% Laplacean
\newcommand{\lap}{\ensuremath{\V^2}}
% Dalembertian
\newcommand{\dal}{\ensuremath{\mathop{}\!\Box}}

% Diff. operators in cartesian coordinates
\newcommand{\grad}{\V}
\newcommand{\rot}{\V\x}
\newcommand{\di}{\V\cdot}

% Diff. operators in any coordinate system
\DeclareMathOperator{\Grad}{grad}
\DeclareMathOperator{\Rot}{rot}
\DeclareMathOperator{\Di}{div}

% Differential symbol 'd'
\newcommand{\ud}{\mathop{}\!\mathrm{d}}
% Partial differential symbol
\newcommand{\de}{\mathop{}\!\mathrm{\partial}}
% Finite difference Delta
\newcommand{\uD}{\mathop{}\!\Delta}
